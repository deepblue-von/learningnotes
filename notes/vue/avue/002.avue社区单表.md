###   davue社区单表

#### vue部分

```vue
<template>
  <basic-container>
    <avue-crud :option="option" //组件配置属性，option改变之后表格发生变化
               :table-loading="loading" //表格加载时触发，[转圈]，loading默认值为true
               :data="data" //表格中所有显示的数据，是一个数组，数组中的每一个对象都是一行数据
               :page="page" //分页，页面信息发生变化时同步到表格
               :permission="permissionList" //权限
               :before-open="beforeOpen" //编辑，查看打开对话框时的回调         
               v-model="form" //对话框中的双向数据绑定，
               ref="crud" //表格的引用，在该组件外部可以通过，子组件的ref引用将会注册到父组件的$refs上
               @row-update="rowUpdate" //一行的数据修改之后触发的事件，编辑事件
               @row-save="rowSave" //新增数据之后触发的事件，左上角的新增按钮
               @row-del="rowDel" //菜单栏的删除执行之后触发的事件
               @search-change="searchChange" //搜索框中的内容发生变化时出发的事件 
               @search-reset="searchReset" //点击清空按钮之后触发的事件
               @selection-change="selectionChange" //表格第一列的选择框状态发生变化时触发的事件
               @current-change="currentChange" //当前页码发生变化时触发的事件
               @size-change="sizeChange" //当前页容纳数据的最大条数变化时触发的事件
               @refresh-change="refreshChange" //点击刷新按钮之后触发的事件
               @on-load="onLoad"> <!--加载页面时触发的事件，效果，数据没有渲染出来时转圈-->
      <template slot="menuLeft">  <!--自定义表格头部左侧的内容-->
        <el-button type="danger" size="small" icon="el-icon-delete" plain 
           v-if="permission.agency_delete" //判断用户是否有删除权限，v-if值为false时，按钮不会显示
           @click="handleDelete">删除</el-button>
      </template>
       <template slot="mapSelectForm" slot-scope="scope">
         <avue-map v-model="scope.row.mapSelect"></avue-map>
      </template>
    </avue-crud>
  </basic-container>
</template>
```
