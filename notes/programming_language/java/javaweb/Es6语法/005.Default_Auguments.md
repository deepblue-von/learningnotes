### 函数参数的默认值

```js
    function multiply(a, b) {
        a = a || 5;
        b = b || 6;
        return a * b;
    }
 // Es6语法
    function multiply(a = 5, b = 3) {
        return a * b;
    }
```

```js
multiply(3); // 9
multiply(,3); // error
multiply(undefined, 3) // 15  typeof a===undefined

```
