### jdbc

#### 基础方法

**1. 概念**： Java Database Connectivity

​       **本质**： 一套操作所有关系型数据库的规则，即接口。

**2. 快速入门 **：

   1. 导入驱动jar包 mysql-connector-java-8.0.19.jar

      * 复制jar包到libs目录下
      * 右键-add as library

   2. 注册驱动

   3. 获取数据库连接 connection

   4. 定义sql

   5. 获取执行sql语句的对象 Statement

   6. 执行sql,接受返回结果

   7. 处理结果

   8. 释放资源

```java
public class JdbcDemo2 {
    public static void main(String[] args) {
        Statement stmt = null;
        Connection conn = null;
        try {
            // 1. 注册驱动
            Class.forName("com.mysql.cj.jdbc.Driver");
            // 2. 定义sql
            String sql = "insert into account values(null,'yangwu',3000)";
            // 3. 获取Connection对象
            conn = DriverManager.getConnection("jdbc:mysql:///db3","root", "root");
            // 4. 获取执行sql的对象
            stmt = conn.createStatement();
            // 5. 执行sql
            int count = stmt.executeUpdate(sql);
            // 6. 处理结果
            System.out.println(count);
            if(count > 0) {
                System.out.println("Successfully added!");
            }else {
                System.out.println("Addition Failed!");
            }
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        } catch (SQLException throwables) {
            throwables.printStackTrace();
        }finally {
            // stmt.close();
            // 出现错误后，会直接跳到catch里面，stmt有可能为null，执行close会出现空指针异常
            
            // 7. 释放资源
            if(stmt != null) {
                try {
                    stmt.close();
                } catch (SQLException throwables) {
                    throwables.printStackTrace();
                }
            }
            if(conn != null) {
                try {
                    conn.close();
                } catch (SQLException throwables) {
                    throwables.printStackTrace();
                }
            }
        }
    }
}

```
**3. 详解各个对象**：  

1. DriverManager: 驱动管理对象
 * 注册驱动: 告诉程序该使用哪一个jar包 (mysql5之后可以不用写注册驱动)
 * 获取数据库连接：getConnection()
   * url:指定链接路径 jdbc:mysql:///db3（连接自己的数据库时的简写）
   * user：用户名
   * password: 密码
2. Connection：数据库链接对象

   * Statement createStatement()

   * PreparedStatement prepareStatement(String sql) 

     管理事务：开启：setAutoCommit(false)

     ​				   提交：commit()

     ​				   回滚：roolback()

3. Statement: 执行sql的对象

   * 返回值是int，表示影响的行数

4. ResultSet: 结果集对象

   * next() 游标向下移动一行
   * getXXX(列名或序号) 获取某一行某一列的数据 

* 正确用法

  **遍历结果集**

  1. 游标向下移动一行
  2. 判断是否有数据
  3. 获取数据

5. PrepareStatement: 执行sql的对象-----解决sql注入，

    **预编译sql**：sql的参数用？作为占位符

   `select * from user where username = ? and password = ?`

   PrepareStatement


#### 将数据封装为对象

```java
/**
 * 定义一个方法，查询stu表的数据将其封装为对象，然后装载集合，返回
 */
public class JdbcDemo8 {
    public static void main(String[] args) {
        List<Student> list = new JdbcDemo8().findAll();
        System.out.println(list);
    }

    /**
     * 查询所有student对象
     * @return
     */
    public List<Student> findAll() {

        Connection conn = null;
        Statement stmt = null;
        ResultSet rs = null;
        List<Student> list = null;
        try {
            // 1. 注册驱动
            Class.forName("com.mysql.cj.jdbc.Driver");
            // 2. 获取连接
            conn = DriverManager.getConnection("jdbc:mysql:///db3", "root", "root");
            // 3. 定义sql
            String sql = "select * from stu";
            // 4. 获取执行sql的对象
            stmt = conn.createStatement();
            // 5. 执行sql
            rs = stmt.executeQuery(sql);
            // 6. 遍历
            Student stu = null;
            list = new ArrayList<Student>();
            while(rs.next()) {
                // 获取数据
                int id = rs.getInt("id");
                String name = rs.getString("name");
                double score = rs.getDouble("score");
                Timestamp insert_time = rs.getTimestamp("insert_time");
                // 创建Student对象，并赋值
                stu = new Student();
                stu.setId(id);
                stu.setName(name);
                stu.setScore(score);
                stu.setInsert_time(insert_time);

                // 装载集合
                list.add(stu);
            }
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        } catch (SQLException throwables) {
            throwables.printStackTrace();
        }finally {
            if(rs != null) {
                try {
                    rs.close();
                } catch (SQLException throwables) {
                    throwables.printStackTrace();
                }
            }
            if(stmt != null) {
                try {
                    stmt.close();
                } catch (SQLException throwables) {
                    throwables.printStackTrace();
                }
            }
            if(conn != null) {
                try {
                    conn.close();
                } catch (SQLException throwables) {
                    throwables.printStackTrace();
                }
            }
        }

        return list;
    }
}
```

```java
   /**
     * 文件的读取只需要一次
     * 使用静态代码块实现
     */
    static {
        try {
            // 获取资源文件，获取值
            // 1. Properties 集合类
            Properties pro = new Properties();
            // 获取src路径下文件的方式----->ClassLoader  类加载器
            ClassLoader classLoader = JDBCUtils.class.getClassLoader();
            URL res = classLoader.getResource("jdbc.properties");
            String path = res.getPath();
            System.out.println(path);
            // 2. 加载文件
            pro.load(new FileReader(path));
            // 3. 获取数据，赋值
            url = pro.getProperty("url");
            user = pro.getProperty("user");
            password = pro.getProperty("password");
            driver = pro.getProperty("driver");
            // 4.注册驱动
            Class.forName(driver);
        } catch (IOException e) {
            e.printStackTrace();
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }
    }

    /**
     * 获取连接对象
     * @return 连接对象
     */
    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(url, user, password);
    }
    /**
     * 释放资源
     * @param stmt
     * @param conn
     */
    public static void close(Statement stmt, Connection conn) {
        if(stmt != null) {
            try {
                stmt.close();
            } catch (SQLException throwables) {
                throwables.printStackTrace();
            }
        }
        if(conn != null) {
            try {
                conn.close();
            } catch (SQLException throwables) {
                throwables.printStackTrace();
            }
        }
    }
    public static void close( ResultSet rs, Statement stmt, Connection conn) {
        if(rs != null) {
            try {
                rs.close();
            } catch (SQLException throwables) {
                throwables.printStackTrace();
            }
        }
        if(stmt != null) {
            try {
                stmt.close();
            } catch (SQLException throwables) {
                throwables.printStackTrace();
            }
        }
        if(conn != null) {
            try {
                conn.close();
            } catch (SQLException throwables) {
                throwables.printStackTrace();
            }
        }
    }
}
```
