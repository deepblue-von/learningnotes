### class继承

```js
class Student{
    constructor(name) {
        this.name = name;
    }
    
    hello() {
        alert('hello');
    }
}
class PrimaryStudent extends Student {
    constructor(name, grade) {
        super(name);
        this.grade = grade;
    }
    myGrade() {
        alert('I am a primary student.');
    }
}

var shy = new Student('石原里美');
var pstu = new PrimaryStudent('iu',1);
</script>
```
